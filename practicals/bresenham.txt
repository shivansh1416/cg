import matplotlib.pyplot as plt

def bresenham_line(x0, y0, x1, y1):
    # Calculate the differences between the points
    dx = abs(x1 - x0)
    dy = abs(y1 - y0)
    
    # Determine the direction of the increments
    sx = 1 if x0 < x1 else -1
    sy = 1 if y0 < y1 else -1
    
    err = dx - dy
    
    # Create lists to store the points
    x_points = []
    y_points = []
    
    # Generate points along the line
    while True:
        x_points.append(x0)
        y_points.append(y0)
        
        if x0 == x1 and y0 == y1:
            break
        
        e2 = 2 * err
        if e2 > -dy:
            err -= dy
            x0 += sx
        if e2 < dx:
            err += dx
            y0 += sy
    
    return x_points, y_points

# Example usage of Bresenham's algorithm
x0, y0 = 2, 3
x1, y1 = 10, 8
x_points, y_points = bresenham_line(x0, y0, x1, y1)

# Plotting the points generated by Bresenham's algorithm
plt.plot(x_points, y_points, marker='o')
plt.title('Bresenham\'s Line Drawing Algorithm')
plt.xlabel('X')
plt.ylabel('Y')
plt.grid()
plt.show()
